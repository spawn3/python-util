\begin{description}

  \item[Village] A small collection of dwelling places.  Examples:
    \begin{enumerate}

      \item Marlingford

      \item Saxlingham Nethergate

    \end{enumerate}

  \item[Town] A large collection of dwelling places.  Examples:
    \begin{enumerate}

      \item Great Yarmouth

      \item Beccles

    \end{enumerate}

  \item[City] A large town, usually containing a cathedral.  Examples:
    \begin{enumerate}

      \item Norwich

      \item Birmingham

      \item London

    \end{enumerate}

\end{description}

\end{document}
